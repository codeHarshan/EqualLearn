<script>
    document.getElementById("otp").style.visibility="hidden";
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    const recognition = new SpeechRecognition();
    let recognitionActive = false;

    // Function to start speech recognition
    function startRecognition() {
        if (!recognitionActive) {
            recognition.start();
            recognitionActive = true;
        }
    }

    recognition.onresult = function(event) {
        const transcript = event.results[0][0].transcript;
        console.log('You said: ' + transcript);
        const emailInput = document.getElementById("email");
        const passwordInput = document.getElementById("password");

        // Regular expressions to match "my email is [email]" and "my password is [password]"
        const emailRegex = /my email is (.+)/i;
        const passwordRegex = /my password is (.+)/i;

        // Check if the transcript matches the email pattern
        const emailMatch = transcript.match(emailRegex);
        if (emailMatch) {
            let email = emailMatch[1];
            email = email.replace(/\sat\s/i, "@").trim(); // Replace "at" with "@" and trim whitespace
            emailInput.focus(); // Activate the email input field
            emailInput.value = email.toLowerCase(); // Set the recognized email in lowercase as the value of the email input
            passwordInput.value = ''; // Clear the password input
            recognition.stop(); // Stop recognition temporarily
            recognitionActive = false; // Reset recognition status
            setTimeout(startRecognition, 500); // Restart recognition after a delay
            return; // Exit the function to prevent setting the password again
        }

        // Check if the transcript matches the password pattern
        const passwordMatch = transcript.match(passwordRegex);
        if (passwordMatch) {
            const password = passwordMatch[1];
            passwordInput.focus(); // Activate the password input field
            passwordInput.value = password.trim(); // Set the recognized password as the value of the password input
            recognitionActive = false; // Reset recognition status
        }
    };

    recognition.onerror = function(event) {
        console.error('Speech recognition error detected: ' + event.error);
        recognitionActive = false; // Reset recognition status if an error occurs
    };

    // Start speech recognition when the page loads
    window.addEventListener("DOMContentLoaded", startRecognition);

    // Start speech recognition when the email input field is focused
    document.getElementById("email").addEventListener("focus", startRecognition);

    // Toggle password visibility when the "Show Password" checkbox is clicked
    document.getElementById("showPassword").addEventListener("change", function() {
        const passwordInput = document.getElementById("password");
        passwordInput.type = this.checked ? "text" : "password";
    });

    // Define loginBtn variable
    const loginBtn = document.getElementById("loginBtn");

    // Handle login button click event
    loginBtn.addEventListener("click", function() {
        const email = document.getElementById("email").value;
        const password = document.getElementById("password").value;
        // Add your login logic here
        console.log("Email:", email);
        console.log("Password:", password);
        // Example: perform login validation or submit form
    });

</script>
